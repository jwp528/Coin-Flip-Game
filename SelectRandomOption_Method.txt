    private void SelectRandomOption()
    {
        if (selectingFor == "heads")
        {
            isHeadsRandom = true;
            // Set a placeholder image for the counter icon
            selectedHeadsImage = "/img/coins/logo.png"; // Default fallback
        }
        else if (selectingFor == "tails")
        {
            isTailsRandom = true;
            // Set a placeholder image for the counter icon
            selectedTailsImage = "/img/coins/logo.png"; // Default fallback
        }
        
        CloseCoinSelector();
    }
